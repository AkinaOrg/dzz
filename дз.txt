# Запитати логін і пароль від першого користувача
login1 = input("Введіть логін першого користувача: ")
password1 = input("Введіть пароль першого користувача: ")

# Запитати логін і пароль від другого користувача
login2 = input("Введіть логін другого користувача: ")
password2 = input("Введіть пароль другого користувача: ")

# Створити список з логіном і паролем кожного користувача
user1 = [login1, password1]
user2 = [login2, password2]

# Порівняти довжину паролів
if len(password1) > len(password2):
    # Якщо пароль першого користувача довший, то видаляємо логін і пароль другого користувача
    del user2[:]
else:
    # В іншому випадку видаляємо логін і пароль першого користувача
    del user1[:]

# Вивести список з логіном і паролем користувача, у якого пароль залишився
if user1:
    print(user1)
else:
    print(user2)
